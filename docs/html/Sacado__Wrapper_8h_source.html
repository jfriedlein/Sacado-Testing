107  /*
108  * Assemble the tangent based on the derivatives saved in the argument sigma
109  * @param Tangent The desired tangent that will be computed as d_sigma/d_this, where "this" could be your strain tensor
110  * @param sigma The input argument used to extract the derivatives
111  */
112  template<int dim>
113  void SymTensor<dim>::get_tangent( SymmetricTensor<4,dim> &Tangent, SymmetricTensor<2,dim, fad_double> &sigma )
114  {
115  for ( unsigned int i=0; i<dim; ++i)
116  for ( unsigned int j=0; j<dim; ++j )
117  {
118  double *derivs = &sigma[i][j].fastAccessDx(0); // Access the derivatives of the (i,j)-th component of \a sigma
119 
120  // We loop over all the dofs. To be able to use this independent of the chosen dimension \a dim, we use a ternary operator
121  // to decide whether we have to loop over 6 derivatives or just 3.
122  for(unsigned int x=start_index;x<(start_index+n_dofs);++x)
123  {
124  unsigned int k=std_map_indicies[x].first;
125  unsigned int l=std_map_indicies[x].second;
126 
127  if(k!=l)/*Compare to Voigt notation since only SymmetricTensor instead of Tensor*/
128  {
129  Tangent[i][j][k][l] = 0.5*derivs[x];
130  Tangent[i][j][l][k] = 0.5*derivs[x];
131  }
132  else
133  Tangent[i][j][k][l] = derivs[x];
134  }
135  }
136  }
