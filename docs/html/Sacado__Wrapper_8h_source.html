127  void SymTensor<dim>::get_tangent( SymmetricTensor<4,dim> &Tangent, SymmetricTensor<2,dim, fad_double> &sigma )
128  {
129  for ( unsigned int i=0; i<dim; ++i)
130  for ( unsigned int j=0; j<dim; ++j )
131  {
132  double *derivs = &sigma[i][j].fastAccessDx(0); // Access the derivatives of the (i,j)-th component of \a sigma
133 
134  // We loop over all the dofs. To be able to use this independent of the chosen dimension \a dim, we use a ternary operator
135  // to decide whether we have to loop over 6 derivatives or just 3.
136  for(unsigned int x=start_index;x<(start_index+n_dofs);++x)
137  {
138  unsigned int k=std_map_indicies[x].first;
139  unsigned int l=std_map_indicies[x].second;
140 
141  if(k!=l)/*Compare to Voigt notation since only SymmetricTensor instead of Tensor*/
142  {
143  Tangent[i][j][k][l] = 0.5*derivs[x];
144  Tangent[i][j][l][k] = 0.5*derivs[x];
145  }
146  else
147  Tangent[i][j][k][l] = derivs[x];
148  }
149  }
